Calendar =
VAR MinDate = MIN('Adherence'[LocalDate])
VAR MaxDate = MAX('Adherence'[LocalDate])
RETURN
ADDCOLUMNS(
    CALENDAR(MinDate, MaxDate),
    "Year", YEAR([Date]),
    "Month", FORMAT([Date], "MMM"),
    "MonthNum", MONTH([Date]),
    "WeekNum", WEEKNUM([Date], 2),          -- 2 = week starts Monday
    "Day", DAY([Date]),
    "Dow", FORMAT([Date], "ddd")
)




Agent =
SUMMARIZE(
    'Adherence',
    'Adherence'[AgentId],
    'Adherence'[AgentName],
    'Adherence'[Team],
    'Adherence'[ManagementUnit]
)
